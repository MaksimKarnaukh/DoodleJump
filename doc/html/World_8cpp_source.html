42 void World::checkForCollision()
43 {
44 
45  float x0 = doodle->getPreviousPositionX() + doodle->getWidth() / 2;
46  float y0 = doodle->getPreviousPositionY() - doodle->getHeight();
47  float x1 = doodle->getPositionX() + doodle->getWidth() / 2;
48  float y1 = doodle->getPositionY() - doodle->getHeight();
49 
50  std::vector<std::pair<float, float>> middleLine = getLineBetweenPoints(x0, y0, x1, y1);
51 
52  if ((doodle->getPositionY() <= lowerBound) ||
53  (doodle->getPositionY() <= lowerBound + logic::utility::Camera::Instance().getShiftValue())) {
54  gameOver = true;
55  }
56 
57  for (auto b = 0; b < bonuses.size(); b++) {
58 
59  if (checkForUndetectedCollision(bonuses[b], middleLine, doodle)) {
60  bonuses[b]->touchedBonus(doodle);
61  score->setScore(score->getScore() + bonuses[b]->getScoreIncrease());
62  for (auto i = 0; i < bonuses[b]->getObservers().size(); i++) {
63  bonuses[b]->removeObserver(bonuses[b]->getObservers()[i]); // remove all observers
64  }
65  bonuses.erase(bonuses.begin() + b);
66  break;
67  }
68  }
69 
70  // check if we hit a platform.
71  for (auto pl = 0; pl < platforms.size(); pl++) {
72 
73  if (checkForUndetectedCollision(platforms[pl], middleLine, doodle)) {
74  doodle->jump();
75  float dec = platforms[pl]->isTouched();
76  if ((int)(dec * 10)) { // not temp platform
77  if (platforms[pl]->getTimesTouched() > 1) {
78  score->setScore(score->getScore() - dec);
79  }
80  } else { // temp platform
81  for (auto i = 0; i < platforms[pl]->getObservers().size(); i++) {
82  platforms[pl]->removeObserver(platforms[pl]->getObservers()[i]);

114 void World::createEntities()
115 {
116 
117  // calculate chance of placing a platform or not.
118  float p = platformGen_startingChance; // begin value = 0.90 (90% chance for platform generation)
119  float factor = std::max(
120  0.25f, (1.0f - (std::floor(doodle->getPositionY() / platformGen_interval)) /
121  100)); // represents the depending factor on the score (y-coordinate of our player)
122  bool isCreate = logic::utility::Random::Instance().bernoulliDistribution(p * factor);
123 
124  // y position of our platform
125  float platformYPos = 1 + logic::utility::Camera::Instance().getShiftValue();
126  if (platformYPos > formerPlatformPosY + minDistanceBetweenPlatforms) { // min distance between platforms guaranteed
127  isCreate = true;
128  }
129 
130  if (formerPlatformPosY + 0.05 < platformYPos) {
131 
132  if (isCreate) {
133  formerPlatformPosY = platformYPos;
134 
135  // decide which platform to place
136  int pl = choosePlatformType();
137 
138  std::shared_ptr<logic::Platform> platform = nullptr;
139  std::shared_ptr<logic::Bonus> bonus = nullptr;
140 
141  float platformXPos = 0; // going to be overwritten
142 
143  // calculate chance of placing a bonus or not.
144  float chanceForBonus = bonusGen_startingChance;
145  float factorBonus =
146  std::min(0.08f, (std::floor(doodle->getPositionY() / bonusGen_interval)) /
147  100); // per 500 height, the chance for a bonus gets decreased by 0.01
148  // with at the end a minimum value of 0.02.
149  bool isCreateBonus = false;
150 
151  switch (pl) {
152  case 0:
153  platform = Factory->createStaticPlatform(platformXPos, platformYPos, platformWidth,
154  platformHeight);
155 
156  isCreateBonus = logic::utility::Random::Instance().bernoulliDistribution(
157  chanceForBonus - factorBonus);
158  if (isCreateBonus) {
159  // decide which bonus to place
160  chooseBonusType(bonus);
161  }
162  break;
163  case 1:
164  platform = Factory->createTemporaryPlatform(platformXPos, platformYPos, platformWidth,
165  platformHeight);
166  break;
167  case 2:
168  platform = Factory->createVerticalPlatform(platformXPos, platformYPos, platformWidth,
169  platformHeight);
170  break;
171  case 3:
172  platform = Factory->createHorizontalPlatform(platformXPos, platformYPos, platformWidth,
173  platformHeight);
174  break;
175  default:
176  std::cout << "No platform was created"
177  << std::endl; // default statement used for error output
178  }
179 
180  // calculate where to place the platform (x position only)
181  platformXPos = calculatePlatformXPos(platform);
182 
183  platform->setPositionX(platformXPos);
184  platforms.push_back(platform);
185 
186  formerPlatformPosX = platformXPos;
187 
188  if (isCreateBonus) {
189  float bonusXPos = logic::utility::Random::Instance().uniformRealDistribution(
190  platformXPos + 0.01f,
191  platformXPos + platform->getWidth() - bonus->getWidth() - 0.01f);
192  bonus->setPositionX(bonusXPos);
193  bonus->setPositionY(platformYPos + bonus->getHeight());
194  bonuses.push_back(bonus);
195  wasBonusGenerated = true;
196  } else {
197  wasBonusGenerated = false;
198  }
199  }
200  }

267 std::vector<std::pair<float, float>> World::getLineBetweenPoints(float x0, float y0, float x1, float y1)
268 { // function self-modified for this project
269 
270  std::vector<std::pair<float, float>> line;
271  float precisionOfIterator = 0.001f;
272 
273  if (x0 == x1) {
274  // special case for x0 == x1
275  for (float i = std::min(y0, y1); i <= std::max(y0, y1); i += precisionOfIterator) {
276  line.emplace_back(x0, i);
277  }
278  } else if (y0 == y1) {
279  // special case for y0 == y1
280  for (float i = std::min(x0, x1); i <= std::max(x0, x1); i += precisionOfIterator) {
281  line.emplace_back(i, y0);
282  }
283  } else {
284  if (x0 > x1) {
285  // flip points if x1<x0: we want x0 to have the lowest value
286  std::swap(x0, x1);
287  std::swap(y0, y1);
288  }
289 
290  float m = (y1 - y0) / (x1 - x0);
291  if (-1.0 <= m && m <= 1.0) {
292  for (float i = 0; i <= (x1 - x0); i += precisionOfIterator) {
293  line.emplace_back(x0 + i, (y0 + m * i));
294  }
295  } else if (m > 1.0) {
296  for (float i = 0; i <= (y1 - y0); i += precisionOfIterator) {
297  line.emplace_back((x0 + (i / m)), y0 + i);
298  }
299  } else if (m < -1.0) {
300  for (float i = 0; i <= (y0 - y1); i += precisionOfIterator) {
301  line.emplace_back((x0 - (i / m)), y0 - i);
302  }
303  }
304  }
305  return line;
306 }

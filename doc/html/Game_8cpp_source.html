107 void Game::playGame()
108 {
109 
110  cFactory = std::move(std::make_shared<representation::ConcreteFactory>());
111  factory = cFactory;
112 
113  world = std::move(std::make_unique<logic::World>(factory));
114 
115  logic::utility::Stopwatch::Instance().Reset();
116 
117  while (!(world->gameOver || GetWindow().IsDone())) {
118 
119  logic::utility::Stopwatch::Instance().tick();
120 
121  if (logic::utility::Stopwatch::Instance().getDeltaTime() < (1.0f / getFrameRate())) {
122  std::chrono::milliseconds ms = std::chrono::milliseconds(
123  (int)(((1.0f / getFrameRate()) - logic::utility::Stopwatch::Instance().getDeltaTime()) *
124  1000));
125  std::this_thread::sleep_for(ms);
126  }
127 
128  logic::utility::Stopwatch::Instance().tick(); // de verstreken milliseconden erbij.
129  // std::cout << 1 / logic::utility::Stopwatch::Instance().getDeltaTime() << std::endl;
130  logic::utility::Stopwatch::Instance().Reset(); // mStartTime = now (tijd verstreken is terug 0)
131 
132  // Game loop.
133  HandleInput();
134  Update();
135  Render();
136  }
137  gameState = 0;
138  calculateHighScore();
139 }
